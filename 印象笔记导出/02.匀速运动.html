02.匀速运动

  • 速度定义一个确切值
  • 开启定时器，定时获取每个时间段的当前值
    • 定时获取每个时间段的当前值
    • 把当前值加上速度进行改变
    • 将改变后的值复制给元素的样式
  • 判断改变后的当前值大于目标值。
    • 判断放在赋值样式前

匀速运动

速度保持不变的运动
案例：飞翔的小鸟

<script type="text/javascript">
        document.addEventListener("DOMContentLoaded", ()=>{
            var bird = document.querySelector("#bird");
            var speed = 5;
            bird.timer = setInterval(()=>{
                var currentLeft = bird.offsetLeft;
                currentLeft += speed;
                if(currentLeft >= window.innerWidth-bird.offsetWidth){
                    currentLeft = window.innerWidth-bird.offsetWidth;
                    clearInterval(bird.timer);
                }
                bird.style.left = currentLeft + "px";
            }, 30)
            // 一.动画
            // 利用定时器，定时改变属性值，产生动画效果
            // 一.匀速运动
            //  1.速度定义一个确切值
            //  2.开启定时器
            //      (1)定时获取每个时间段的当前值
            //      (2)把当前值加上速度进行改变
            //      (3)将改变后的值赋值给元素的样式
            //  3.判断改变后的当前值大于目标值。判断肯定要放在赋值样式前。
        })

    </script>
案例：圆周运动
<script>
        document.addEventListener('DOMContentLoaded',()=>{
            // 页面布局完成：
            // 1.太阳定位在浏览器可视区域中间
            // 2.太阳半径130+25=155  地球半径15px.从而实现地球的定位
            // ==========================================
            //  1.速度定义一个确切值
            //  2.开启定时器
            //      (1)定时获取每个时间段的当前值
            //      (2)把当前值加上速度进行改变
            //      (3)将改变后的值赋值给元素的样式
            //  3.判断改变后的当前值大于目标值。判断肯定要放在赋值样式前。
            let sun = document.querySelector('.sun');
            let earth = document.querySelector('.earth');
            sun.style.left = (window.innerWidth-sun.offsetWidth)/2 + 'px';
            sun.style.top = (window.innerHeight-sun.offsetHeight)/2 + 'px';
            let sun_r = 155;
            let earth_r = 15;
            //初始角度跟速度
            let deg = 0;
            setEarthPos(deg);
            let speed = 5;
            setInterval(()=>{
                deg += speed;
                setEarthPos(deg);
                
            }, 30)
            
            function toRad(deg){
                return deg*Math.PI/180;
            }
            function setEarthPos(deg){
                var rad = toRad(deg);
                var a = sun_r*Math.sin(rad);
                var b = sun_r*Math.cos(rad);
                earth.style.left = window.innerWidth/2+a-earth_r +'px';
                earth.style.top =  window.innerHeight/2-b-earth_r +'px';
            }
        });
    </script>
